impl Expr for u32 {
    fn lit(i: u32) -> Self {
        i
    }
    fn add(lhs: Self, rhs: Self) -> Self {
        lhs + rhs
    }
    fn sub(lhs: Self, rhs: Self) -> Self {
        lhs - rhs
    }
    fn mul(lhs: Self, rhs: Self) -> Self {
        lhs * rhs
    }
    fn div(lhs: Self, rhs: Self) -> Self {
        lhs / rhs
    }
}

impl Expr for String {
    fn lit(i: u32) -> Self {
        i.to_string()
    }
    fn add(lhs: Self, rhs: Self) -> Self {
        format!("({} + {})", lhs, rhs)
    }
    fn sub(lhs: Self, rhs: Self) -> Self {
        format!("({} - {})", lhs, rhs)
    }
    fn mul(lhs: Self, rhs: Self) -> Self {
        format!("({} * {})", lhs, rhs)
    }
    fn div(lhs: Self, rhs: Self) -> Self {
        format!("({} / {})", lhs, rhs)
    }
}
